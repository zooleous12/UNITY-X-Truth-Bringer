/**

function scanWebsite() {

/**
 * Extract title from HTML
 */
function extractTitle(html) {
  const titleMatch = html.match(/<title[^>]*>([^<]+)<\/title>/i);
  return titleMatch ? titleMatch[1].trim() : 'No title';
}

/**
 * Extract all links from HTML
 */
function extractLinks(html, baseUrl) {
  const links = [];
  const linkPattern = /<a[^>]+href=["']([^"']+)["']/gi;
  let match;
  
  while ((match = linkPattern.exec(html)) !== null) {
    const href = match[1];
    
    // Convert relative URLs to absolute
    try {
      const absoluteUrl = resolveUrl(href, baseUrl);
      if (absoluteUrl) links.push(absoluteUrl);
    } catch (e) {
      // Skip invalid URLs
    }
  }
  
  return [...new Set(links)]; // Remove duplicates
}

/**
 * Resolve relative URLs to absolute
 */
function resolveUrl(href, baseUrl) {
  // Skip anchors, javascript, mailto, tel, etc.
  if (!href || 
      href.startsWith('#') || 
      href.startsWith('javascript:') ||
      href.startsWith('mailto:') ||
      href.startsWith('tel:')) {
    return null;
  }
  
  // Already absolute
  if (href.match(/^https?:\/\//i)) {
    return href;
  }
  
  const base = new URL(baseUrl);
  
  // Protocol-relative URL
  if (href.startsWith('//')) {
    return base.protocol + href;
  }
  
  // Absolute path
  if (href.startsWith('/')) {
    return base.origin + href;
  }
  
  // Relative path
  const basePath = base.pathname.substring(0, base.pathname.lastIndexOf('/') + 1);
  return base.origin + basePath + href;
}

/**
 * Normalize URL (remove fragments, trailing slashes)
 */
function normalizeUrl(url) {
  try {
    const urlObj = new URL(url);
    // Remove fragment
    urlObj.hash = '';
    let normalized = urlObj.toString();
    // Remove trailing slash for consistency
    if (normalized.endsWith('/') && urlObj.pathname !== '/') {
      normalized = normalized.slice(0, -1);
    }
    return normalized;
  } catch (e) {
    return url;
  }
}

/**
 * Determine if a URL should be crawled
 */
function shouldCrawl(url, startUrl) {
  try {
    const urlObj = new URL(url);
    const startUrlObj = new URL(startUrl);
    
    // Check same domain if configured
    if (CONFIG.SAME_DOMAIN_ONLY) {
      if (urlObj.hostname !== startUrlObj.hostname) {
        return false;
      }
    }
    
    // Skip common non-HTML files
    const path = urlObj.pathname.toLowerCase();
    const skipExtensions = ['.pdf', '.jpg', '.jpeg', '.png', '.gif', '.css', '.js', '.xml', '.zip', '.exe'];
    if (skipExtensions.some(ext => path.endsWith(ext))) {
      return false;
    }
    
    return true;
  } catch (e) {
    return false;
  }
}

/**
 * Calculate URL depth relative to start URL
 */
function getDepth(url, startUrl) {
  try {
    const urlPath = new URL(url).pathname;
    const startPath = new URL(startUrl).pathname;
    const depth = urlPath.split('/').length - startPath.split('/').length;
    return Math.max(0, depth);
  } catch (e) {
    return 0;
  }
}
